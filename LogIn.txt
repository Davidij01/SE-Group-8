   public boolean logIn(String passWord) {
       String info = "";
       String userAdmin = "user"; // default assume user so as not to accidentally allow privileged access
       
       if (passWord.contains("-")) {
           System.out.println("User");
           info = getUserInfo(passWord, true);
           userAdmin = "user";
       } else {
           System.out.println("Admin");
           info = getAdminInfo(passWord);
           userAdmin = "admin";
       }
       
       if (info.compareTo("UserNotFound") != 0) {
         newTempFile();
         
         // Allow appropriate privileges
         if (userAdmin.compareTo("user") == 0) {
             isAdmin = false;
             viewAllAccounts(passWord, true, false, false);
         } else {
             isAdmin = true;
         }
         
         return true; // login successful
       } else {
         return false; // login failed
       }
   }
